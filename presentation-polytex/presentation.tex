\begin{untransparent}{PolyTeX, un environnement éditorial}
\vfill
\polytex : \textcolor{yellow}{chaîne éditoriale} de production de
cours matérialisés sur des supports électroniques ou physiques.
\vfill
\begin{itemize}
\item un éditeur de \emph{texte structuré} mais \emph{non formatté} 
\item un \textcolor{yellow}{processeur de texte}, \textcolor{green}{\LaTeX} ou \pdflatex, utilisant le style \textcolor{green}{PolyTeX}
\vfill
\textcolor{yellow}{options} : impression ou écran, polycopié ou transparents
\vfill      
\item \textcolor{yellow}{Un outil de visualisation} (papier, \textcolor{green}{Acrobat Reader}, Browser)
\end{itemize}
\vfill
\end{untransparent}
%
\begin{untransparent}{Les différentes plate-formes}
\vfill
PolyTeX est disponible sous :
\vfill
\begin{itemize}
\item Unix/linux, distribution \green{TeTeX + nedit}

\item MacOs, distribution \green{OzTeX + Alpha}
\end{itemize}
\vfill
\begin{center}
\emph{http://www.ema.utc.fr/polytex/}

\vfill
\end{untransparent}

\begin{untransparent}{Historique - points forts / points faibles}
\vfill
Points forts : 
\vfill
\begin{itemize}
\item L'auteur a un travail de rédaction et de structuration et non de mise en forme
\item Elaboration rapide de nouveaux Thèmes
\item Evolutivité (XML,...)
\end{itemize}
\vfill
Points faibles :
\vfill
\begin{itemize}
\item Pas d'éditeur dédié.
\item Pas de prise en charge de la gestion et de la cohérence des textes produits.
\end{itemize}
\end{untransparent}
